    /**
     * Restarts the current Rest driver
     * @param {Function} done A callback for completion
     * @returns {RestDriver} The current RestDriver instance
     */
    this.restart = function (done) {
        done = arguments.sodalast() instanceof Function ? arguments.sodalast() : function () {};

        if(!lastInitOptions) {
            done.call(self, new Error("Cannot restart Rest framework, since it hasn't been started yet"), false, null);
        }
        else {
            self.stop(() => { self.start(lastInitOptions, done); });
        }
        return self;
    };

    /**
     * Sends a post command to the a uri with headers and a body
     * @param {string} uri The uri to post to
     * @param {string} headers The headers to send along with the post
     * @param {string} body The body to send along with the post
     * @param {Function} done A callback for completion
     */
    this.post = function(uri, headers, body, done) {
      var t    = new Date().getTime();

      request.post(
             {
               uri: uri,
               body: body,
               headers: headers,
               proxy: soda.config.get("proxy"),
               timeout: 15000
             },
           function (err, response, body) {
               var statusCode = 999,
                   statusMessage = 'Issue with request';

               if(response && response.statusCode) {
                 statusCode = response.statusCode;
               }
               if(response && response.statusMessage) {
                 statusMessage = response.statusMessage;
               }

                var finalResult = {
                  statusCode: statusCode,
                  statusMessage: statusMessage,
                  timeStamp: t,
                  uri: uri
                };

                if (err) {
                  finalResult.errorCode = 'GENERROR';
                  finalResult.errorDescription = err.message;
                  finalResult.passfail = 0;

                  done(err, finalResult, response);
                }

                try {
                  var jsonResult = JSON.parse(body);

                  if (jsonResult.errors) {
                    finalResult.errorCode = jsonResult.errors[0].errorCode;
                    finalResult.errorDescription = jsonResult.errors[0].errorDescription;
                    finalResult.passfail = 0;

                    tree = jsonResult;

                    done(new Error("Error received from API: " + body), finalResult, response);
                  }
                  else {
                    finalResult.errorCode = '';
                    finalResult.errorDescription = '';
                    finalResult.passfail = 1;
                  }

                  tree = jsonResult;

                  done(null, finalResult, response);
                }
                catch (e) {
                  var errorCode = 'EXCEPTION';
                  var finalBody = e.message;
                  if (body) {
                    finalBody = body.match(/<title[^>]*>([^<]+)<\/title>/)[1];
                    errorCode = finalBody.split(':')[0];
                  }
                  else if (e.message.match(/Unexpected token/)) {
                    finalBody = 'No data in response';
                  }
                  finalResult.errorCode = errorCode;
                  finalResult.errorDescription = finalBody;
                  finalResult.passfail = 0;

                  tree = {};

                  done(null, finalResult, response);
                }
            });
    };
